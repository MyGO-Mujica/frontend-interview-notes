#### 1.计数器
原生
```javascript
<div>
  <p id="count">0</p>
  <button id="inc">+1</button>
  <button id="dec">-1</button>
</div>

<script>
  let count = 0
  const countEl = document.getElementById('count')
  const incBtn = document.getElementById('inc')
  const decBtn = document.getElementById('dec')

  function render() {
    countEl.textContent = count 
  }
  incBtn.addEventListener('click', () => {
    count++
    render()
  })
  decBtn.addEventListener('click', () => {
    count = Math.max(0,count - 1)
    render()
  })
  render()
</script>
```
hooks
```javascript
import { useState } from "react";
export default function Counter() {
  const [count, setCount] = useState(0)

  return(
    <div>
      <p>{count}</p>
      <button onClick={() => setCount(c => c+ 1)}>+1</button>
      <button onClick={() => setCount(c => Math.max(0,c-1))}>-1</button>
    </div>
  )
}
```

hooks
```javascript
import { useRef, useEffect, useCallback } from 'react'

function useDebounce(fn, delay = 300) {
  const timer = useRef(null)
  // 用 useRef 保存最新的函数，避免闭包陷阱
  const fnRef = useRef(fn)

  // 每次 fn 变化，更新最新的函数引用
  useEffect(() => {
    fnRef.current = fn
  }, [fn])

  // 组件卸载时，清除定时器，防止内存泄漏
  useEffect(() => {
    return () => {
      clearTimeout(timer.current)
    }
  }, [])

  // 用 useCallback 缓存函数，避免不必要的重渲染
  // useCallback 不是优化函数执行速度，而是固定函数的内存地址，
  return useCallback((...args) => {
    if (timer.current) clearTimeout(timer.current)
    
    timer.current = setTimeout(() => {
      fnRef.current(...args)
    }, delay)
  }, [delay])
}

//这个 debounce hook 的核心思路是通过 useRef 保存 timer，实现延迟执行，同时通过 useCallback 保证返回函数稳定。
//为了解决闭包问题，我用 useRef 保存最新的 fn，并在 useEffect 中同步更新。
//同时在组件卸载时清理定时器，避免内存泄漏。
```

**防抖**
```javascript
function throttle(fn,delay) {
  let lastTime = 0
  return function (...args) {
    const now = Date.now()
    if(now - lastTime > delay){
      lastTime = now
      fn.apply(this,args)
    }
  }
}
```
